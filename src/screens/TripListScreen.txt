import React, { useState, useMemo } from 'react';
import { StyleSheet, Text, View, TextInput, TouchableOpacity, ScrollView } from 'react-native';
import { Trip, TripState } from '../types/trip';
import { TripCard } from '../components/TripCard';

interface TripListScreenProps {
  trips: Trip[];
  onSelectTrip: (id: string) => void;
  onDuplicateTrip: (trip: Trip) => void;
  onDeleteTrip: (id: string) => void;
  onOpenForm: (trip?: Trip) => void;
}

export const TripListScreen: React.FC<TripListScreenProps> = ({
  trips,
  onSelectTrip,
  onDuplicateTrip,
  onDeleteTrip,
  onOpenForm,
}) => {
  const [searchQuery, setSearchQuery] = useState('');
  const [selectedStatusFilter, setSelectedStatusFilter] = useState<TripState | 'tutti'>('tutti');

  const filteredTrips = useMemo(() => {
    return trips.filter((trip) => {
      const matchesSearch =
        trip.titolo.toLowerCase().includes(searchQuery.toLowerCase()) ||
        trip.destinazione.toLowerCase().includes(searchQuery.toLowerCase());
      const matchesStatus =
        selectedStatusFilter === 'tutti' || trip.stato === selectedStatusFilter;
      return matchesSearch && matchesStatus;
    });
  }, [trips, searchQuery, selectedStatusFilter]);

  return (
    <View style={styles.container}>
      <View style={styles.header}>
        <Text style={styles.headerTitle}>Travel Organizer</Text>
        <Text style={styles.headerSubtitle}>Modalità Offline</Text>
      </View>

      <TextInput
        style={styles.searchBar}
        placeholder="Cerca per titolo o destinazione..."
        placeholderTextColor="#888"
        value={searchQuery}
        onChangeText={setSearchQuery}
      />

      <View style={styles.chipContainer}>
        {(['tutti', 'futuro', 'in corso', 'completato'] as const).map((filter) => (
          <TouchableOpacity
            key={filter}
            style={[styles.chip, selectedStatusFilter === filter && styles.chipActive]}
            onPress={() => setSelectedStatusFilter(filter)}
          >
            <Text style={[styles.chipText, selectedStatusFilter === filter && styles.chipTextActive]}>
              {filter.toUpperCase()}
            </Text>
          </TouchableOpacity>
        ))}
      </View>

      <ScrollView contentContainerStyle={styles.scrollList}>
        {filteredTrips.length === 0 ? (
          <Text style={styles.emptyText}>Nessun viaggio trovato.</Text>
        ) : (
          filteredTrips.map((trip) => (
            <TripCard
              key={trip.id}
              trip={trip}
              onPress={() => onSelectTrip(trip.id)}
              onDuplicate={() => onDuplicateTrip(trip)}
              onEdit={() => onOpenForm(trip)}
              onDelete={() => onDeleteTrip(trip.id)}
            />
          ))
        )}
      </ScrollView>

      <TouchableOpacity style={styles.fab} onPress={() => onOpenForm()}>
        <Text style={styles.fabText}>+</Text>
      </TouchableOpacity>
    </View>
  );
};

const styles = StyleSheet.create({
  container: { 
    flex: 1, 
    backgroundColor: '#f5f7fb',
  },
  header: { 
    padding: 20, 
    backgroundColor: '#1e3a8a', 
    borderBottomLeftRadius: 16, 
    borderBottomRightRadius: 16,
  },
  headerTitle: { 
    fontSize: 24, 
    fontWeight: 'bold', 
    color: '#fff',
  },
  headerSubtitle: { 
    fontSize: 14, 
    color: '#93c5fd', 
    marginTop: 2,
  },
  searchBar: { 
    backgroundColor: '#fff', 
    margin: 15, 
    paddingHorizontal: 15, 
    paddingVertical: 10, 
    borderRadius: 10, 
    borderWidth: 1, 
    borderColor: '#e2e8f0', 
    color: '#000',
  },
  chipContainer: { 
    flexDirection: 'row', 
    paddingHorizontal: 15, 
    marginBottom: 10,
  },
  chip: { 
    paddingHorizontal: 12, 
    paddingVertical: 6, 
    borderRadius: 20, 
    backgroundColor: '#e2e8f0', 
    marginRight: 8,
  },
  chipActive: { 
    backgroundColor: '#1e3a8a',
  },
  chipText: { 
    fontSize: 11, 
    fontWeight: '600', 
    color: '#4a5568',
  },
  chipTextActive: { 
    color: '#fff',
  },
  scrollList: { 
    padding: 15,
  },
  emptyText: { 
    textAlign: 'center', 
    color: '#718096', 
    marginTop: 40,
  },
  fab: { 
    position: 'absolute', 
    right: 20, 
    bottom: 20, 
    width: 56, 
    height: 56, 
    borderRadius: 28, 
    backgroundColor: '#1e3a8a', 
    justifyContent: 'center', 
    alignItems: 'center', 
    elevation: 5,
  },
  fabText: { 
    fontSize: 28, 
    color: '#fff', 
    lineHeight: 28,
  },
});
