import React, { useState } from 'react';
import { StyleSheet, Text, View, TouchableOpacity } from 'react-native';
import { Trip } from '../types/trip';

interface TripDetailScreenProps {
  trip: Trip;
  onBack: () => void;
  onDuplicate: () => void;
  onDelete: () => void;
}

export const TripDetailScreen: React.FC<TripDetailScreenProps> = ({
  trip,
  onBack,
  onDuplicate,
  onDelete,
}) => {
  const [activeTab, setActiveTab] = useState<'Itinerario' | 'Checklist' | 'Spese' | 'Info'>('Itinerario');

  return (
    <View style={styles.container}>
      <View style={styles.detailHeader}>
        <TouchableOpacity style={styles.backButton} onPress={onBack}>
          <Text style={styles.backButtonText}>⬅️ Torna</Text>
        </TouchableOpacity>
        <View style={styles.headerTitleContainer}>
          <Text style={styles.detailHeaderTitle} numberOfLines={1}>
            {trip.titolo}
          </Text>
          <Text style={styles.detailHeaderSub}>
            {trip.destinazione}
          </Text>
        </View>
        
        <View style={styles.headerActionGroup}>
          <TouchableOpacity style={styles.iconHeaderBtn} onPress={onDuplicate}>
            <Text>📂</Text>
          </TouchableOpacity>
          <TouchableOpacity style={styles.iconHeaderBtn} onPress={onDelete}>
            <Text>🗑️</Text>
          </TouchableOpacity>
        </View>
      </View>

      <View style={styles.tabBar}>
        {(['Itinerario', 'Checklist', 'Spese', 'Info'] as const).map((tab) => (
          <TouchableOpacity
            key={tab}
            style={[styles.tabItem, activeTab === tab && styles.tabItemActive]}
            onPress={() => setActiveTab(tab)}
          >
            <Text style={[styles.tabText, activeTab === tab && styles.tabTextActive]}>
              {tab}
            </Text>
          </TouchableOpacity>
        ))}
      </View>

      <View style={styles.tabContentContainer}>
        {activeTab === 'Itinerario' && (
          <View style={styles.placeholderTab}>
            <Text style={styles.placeholderTitle}>🗺️ Timeline Itinerario</Text>
            <Text style={styles.placeholderDesc}>
              Gestone Giornate & Attività (Competenza Membro P2/P3)
            </Text>
          </View>
        )}

        {activeTab === 'Checklist' && (
          <View style={styles.placeholderTab}>
            <Text style={styles.placeholderTitle}>✅ Checklist & Documenti</Text>
            <Text style={styles.placeholderDesc}>
              Oggetti da portare e valigie (Competenza Membro P4)
            </Text>
          </View>
        )}

        {activeTab === 'Spese' && (
          <View style={styles.placeholderTab}>
            <Text style={styles.placeholderTitle}>💰 Registro Spese Viaggio</Text>
            <Text style={styles.placeholderDesc}>
              Previsioni vs Spese Effettive (Competenza Membro P5)
            </Text>
            <View style={styles.mockItem}>
              <Text>Budget Totale Assegnato: €{trip.budget}</Text>
            </View>
          </View>
        )}

        {activeTab === 'Info' && (
          <View style={styles.placeholderTab}>
            <Text style={styles.placeholderTitle}>ℹ️ Informazioni Generali & Note</Text>
            <View style={styles.infoNoteBox}>
              <Text style={styles.infoNoteText}>
                {trip.descrizione || "Nessuna nota aggiuntiva."}
              </Text>
            </View>
          </View>
        )}
      </View>
    </View>
  );
};

const styles = StyleSheet.create({
  container: { 
    flex: 1, 
    backgroundColor: '#f5f7fb',
  },
  detailHeader: { 
    flexDirection: 'row', 
    alignItems: 'center',

  },

  },
  backButtonText: { 
    color: '#fff',

    fontSize: 13,
  },
  headerTitleContainer: {
    flex: 1,
    marginLeft: 10,
  },
  detailHeaderTitle: {

    fontWeight: 'bold', 
    color: '#fff',
  },
  detailHeaderSub: { 
    fontSize: 13, 
    color: '#93c5fd',
  },
  headerActionGroup: { 
    flexDirection: 'row',
  },
  iconHeaderBtn: { 
    marginLeft: 12, 
    padding: 8, 
    backgroundColor: '#2563eb', 
    borderRadius: 8,
  },
  tabBar: { 
    flexDirection: 'row', 
    backgroundColor: '#fff', 
    borderBottomWidth: 1, 
    borderBottomColor: '#e2e8f0',
  },
  tabItem: { 
    flex: 1, 
    paddingVertical: 12, 
    alignItems: 'center', 
    borderBottomWidth: 2,

  },
  tabItemActive: { 
    borderBottomColor: '#1e3a8a',
  },
  tabText: {

  },
  placeholderDesc: {

  mockItem: {

  },
  infoNoteBox: {

});
