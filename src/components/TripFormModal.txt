import React, { useState, useEffect } from 'react';
import { StyleSheet, Text, View, TextInput, TouchableOpacity, ScrollView, Modal } from 'react-native';
import { Trip, TripState } from '../types/trip';

interface TripFormModalProps {
  visible: boolean;
  onClose: () => void;
  onSave: (tripData: Omit<Trip, 'id'>) => void;
  editingTrip: Trip | null;
}

export const TripFormModal: React.FC<TripFormModalProps> = ({
  visible,
  onClose,
  onSave,
  editingTrip,
}) => {
  const [titolo, setFormTitolo] = useState('');
  const [destinazione, setFormDestinazione] = useState('');
  const [dataInizio, setFormDataInizio] = useState('');
  const [dataFine, setFormDataFine] = useState('');
  const [descrizione, setFormDescrizione] = useState('');
  const [stato, setFormStato] = useState<TripState>('futuro');
  const [budget, setFormBudget] = useState('');

  useEffect(() => {
    if (editingTrip) {
      setFormTitolo(editingTrip.titolo);
      setFormDestinazione(editingTrip.destinazione);
      setFormDataInizio(editingTrip.dataInizio);
      setFormDataFine(editingTrip.dataFine);
      setFormDescrizione(editingTrip.descrizione);
      setFormStato(editingTrip.stato);
      setFormBudget(editingTrip.budget.toString());
    } else {
      setFormTitolo('');
      setFormDestinazione('');
      setFormDataInizio(new Date().toISOString().split('T')[0]);
      setFormDataFine(new Date().toISOString().split('T')[0]);
      setFormDescrizione('');
      setFormStato('futuro');
      setFormBudget('');
    }
  }, [editingTrip, visible]);

  const handleSave = () => {
    onSave({
      titolo,
      destinazione,
      dataInizio,
      dataFine,
      descrizione,
      stato,
      budget: parseFloat(budget) || 0,
    });
  };

  return (
    <Modal visible={visible} animationType="slide" transparent={true}>
      <View style={styles.modalOverlay}>
        <View style={styles.modalBottomSheet}>
          <Text style={styles.modalTitle}>
            {editingTrip ? 'Modifica Viaggio' : 'Nuovo Viaggio'}
          </Text>
          
          <ScrollView style={styles.formScroll}>
            <Text style={styles.label}>Titolo Viaggio *</Text>
            <TextInput 
              style={styles.input} 
              value={titolo} 
              onChangeText={setFormTitolo} 
              placeholder="Es: Vacanza a Tokyo" 
            />

            <Text style={styles.label}>Destinazione Principale *</Text>
            <TextInput 
              style={styles.input} 
              value={destinazione} 
              onChangeText={setFormDestinazione} 
              placeholder="Es: Giappone" 
            />

            <View style={styles.rowInputs}>
              <View style={styles.flexInputRightMargin}>
                <Text style={styles.label}>Data Inizio (AAAA-MM-GG)</Text>
                <TextInput 
                  style={styles.input} 
                  value={dataInizio} 
                  onChangeText={setFormDataInizio} 
                />
              </View>
              <View style={styles.flexInputLeftMargin}>
                <Text style={styles.label}>Data Fine (AAAA-MM-GG)</Text>
                <TextInput 
                  style={styles.input} 
                  value={dataFine} 
                  onChangeText={setFormDataFine} 
                />
              </View>
            </View>

            <Text style={styles.label}>Budget Previsto (€)</Text>
            <TextInput 
              style={styles.input} 
              value={budget} 
              onChangeText={setFormBudget} 
              keyboardType="numeric" 
            />

            <Text style={styles.label}>Stato del Viaggio</Text>
            <View style={styles.rowInputs}>
              {(['futuro', 'in corso', 'completato'] as const).map((s) => (
                <TouchableOpacity
                  key={s}
                  style={[styles.stateSelector, stato === s && styles.stateSelectorActive]}
                  onPress={() => setFormStato(s)}
                >
                  <Text style={[styles.stateSelectorText, stato === s && styles.stateSelectorTextActive]}>
                    {s}
                  </Text>
                </TouchableOpacity>
              ))}
            </View>

            <Text style={styles.label}>Note / Descrizione</Text>
            <TextInput
              style={[styles.input, styles.textArea]}
              value={descrizione}
              onChangeText={setFormDescrizione}
              multiline
              placeholder="Dettagli utili..."
            />
          </ScrollView>

          <View style={styles.modalButtons}>
            <TouchableOpacity style={[styles.modalBtn, styles.btnCancel]} onPress={onClose}>
              <Text style={styles.btnTextCancel}>Annulla</Text>
            </TouchableOpacity>
            <TouchableOpacity style={[styles.modalBtn, styles.btnSave]} onPress={handleSave}>
              <Text style={styles.btnTextSave}>Salva</Text>
            </TouchableOpacity>
          </View>
        </View>
      </View>
    </Modal>
  );
};

const styles = StyleSheet.create({
  modalOverlay: { 
    flex: 1, 
    backgroundColor: 'rgba(0,0,0,0.5)', 
    justifyContent: 'flex-end',
  },
  modalBottomSheet: { 
    backgroundColor: '#fff', 
    borderTopLeftRadius: 20, 
    borderTopRightRadius: 20, 
    padding: 20, 
    maxHeight: '85%',
  },
  modalTitle: { 
    fontSize: 20, 
    fontWeight: 'bold', 
    color: '#1a202c', 
    marginBottom: 15,
  },
  formScroll: { 
    marginBottom: 20,
  },
  label: { 
    fontSize: 13, 
    fontWeight: '600', 
    color: '#4a5568', 
    marginBottom: 6, 
    marginTop: 10,
  },
  input: { 
    borderWidth: 1, 
    borderColor: '#cbd5e1', 
    borderRadius: 8, 
    padding: 10, 
    fontSize: 15, 
    color: '#000', 
    backgroundColor: '#f8fafc',
  },
  rowInputs: { 
    flexDirection: 'row', 
    justifyContent: 'space-between',
  },
  flexInputRightMargin: {
    flex: 1,
    marginRight: 5,
  },
  flexInputLeftMargin: {
    flex: 1,
    marginLeft: 5,
  },
  stateSelector: { 
    flex: 1, 
    paddingVertical: 8, 
    borderWidth: 1, 
    borderColor: '#cbd5e1', 
    borderRadius: 8, 
    marginRight: 5, 
    alignItems: 'center', 
    backgroundColor: '#f8fafc',
  },
  stateSelectorActive: { 
    backgroundColor: '#1e3a8a', 
    borderColor: '#1e3a8a',
  },
  stateSelectorText: { 
    fontSize: 12, 
    fontWeight: '600', 
    color: '#4a5568',
  },
  stateSelectorTextActive: { 
    color: '#fff',
  },
  textArea: {
    height: 80,
    textAlignVertical: 'top',
  },
  modalButtons: { 
    flexDirection: 'row', 
    justifyContent: 'space-between',
  },
  modalBtn: { 
    flex: 1, 
    padding: 12, 
    borderRadius: 8, 
    alignItems: 'center',
  },
  btnCancel: { 
    backgroundColor: '#e2e8f0', 
    marginRight: 10,
  },
  btnSave: { 
    backgroundColor: '#1e3a8a',
  },
  btnTextCancel: { 
    color: '#4a5568', 
    fontWeight: '600',
  },
  btnTextSave: { 
    color: '#fff', 
    fontWeight: '600',
  },
});
